\begin{frame}{Project Overview}
\textbf{Multi-Piano} is a LAN-based collaborative piano application with a terminal user interface (TUI).

\vspace{0.4cm}
\begin{itemize}
    \item Multiple players on the same network discover, host, and join lobbies.
    \item Notes are transmitted in near real time and played back with \textbf{clock-synchronised} audio.
    \item No external sound assets: piano tones are synthesized procedurally at build or first run.
    \item A 13-key one-octave keyboard (C4--C5) is rendered as an ASCII piano in the terminal.
\end{itemize}

\vspace{0.3cm}
\textbf{Goal:} demonstrate a lightweight multiplayer prototype combining discovery, mixed protocols, NTP-style clock sync, scheduled playout, and a full TUI.
\end{frame}

\begin{frame}{Networking — Three Channels}
We separate concerns into three network channels:

\vspace{0.3cm}
\begin{table}
\centering
\small
\begin{tabular}{clll}
\toprule
\textbf{Port} & \textbf{Protocol} & \textbf{Format} & \textbf{Role} \\
\midrule
12488 & UDP & JSON & Lobby discovery (broadcast) \\
12487 & TCP & length-prefixed JSON & Lobby management \\
12489 & UDP & binary + JSON & Note events \& clock sync \\
\bottomrule
\end{tabular}
\end{table}

\vspace{0.3cm}
\begin{itemize}
    \item \textbf{Discovery} finds open sessions without prior configuration.
    \item \textbf{TCP} maintains player membership and state synchronisation.
    \item \textbf{UDP 12489} carries 15-byte note packets and NTP-style \texttt{SYNC\_REQ}/\texttt{SYNC\_ACK} messages.
\end{itemize}
\end{frame}

\begin{frame}{Lobby Discovery (UDP 12488)}
\textbf{Client behaviour:}
\begin{itemize}
    \item Binds UDP port 12488 with \texttt{SO\_BROADCAST} and \texttt{SO\_REUSEADDR}.
    \item Sends \texttt{DISCOVER} broadcasts; listens for \texttt{LOBBY\_REPLY} responses.
\end{itemize}

\vspace{0.3cm}
\textbf{Host behaviour:}
\begin{itemize}
    \item Replies to discoverers with lobby name, host IP, TCP port, and active player count.
    \item Re-broadcasts \texttt{LOBBY\_REPLY} every 2 seconds.
\end{itemize}

\vspace{0.3cm}
\textbf{Stale lobby eviction:} entries expire after 6 seconds without a reply (cleanup every 3 s).
\end{frame}

\begin{frame}{Lobby Management (TCP 12487)}
All TCP messages use a \textbf{4-byte big-endian length prefix} followed by UTF-8 JSON.

\vspace{0.3cm}
\textbf{Join flow:}
\begin{enumerate}
    \item Client connects to \texttt{(host\_ip, 12487)}.
    \item Client sends \texttt{\{"type": "JOIN", "SENDER\_NAME": ..., "SENDER\_IP": ...\}}.
    \item Host accepts, stores the socket, and appends the client IP to \texttt{peer\_ips}.
    \item Host broadcasts \texttt{STATE\_UPDATE} with the current player list to all clients.
    \item Both sides start a clock-sync loop (every 5 s) over UDP 12489.
\end{enumerate}

\vspace{0.3cm}
Disconnect is detected via TCP EOF; the lobby list is updated and the UI is notified.
\end{frame}

\begin{frame}{Note Streaming \& Clock Synchronisation}
\textbf{Note packets} (15 bytes, little-endian):
\small\texttt{seq | play\_at (float64) | note\_id | action | velocity}

\vspace{0.3cm}
\textbf{Sender scheduling:}
\begin{itemize}
    \item On key press, stamp \texttt{play\_at = time.time() + 30\,ms}.
    \item Local and remote playback both target this absolute time.
\end{itemize}

\vspace{0.3cm}
\textbf{NTP-style clock sync} (JSON on UDP 12489):
\begin{itemize}
    \item \texttt{SYNC\_REQ} carries $t_1$; \texttt{SYNC\_ACK} returns $t_1, t_2, t_3$.
    \item Offset: $\text{offset} = \dfrac{(t_2 - t_1) + (t_3 - t_4)}{2}$ (peer clock $-$ local clock).
    \item Receiver converts: \texttt{local\_play\_at = play\_at $-$ offset}.
\end{itemize}
\end{frame}

\begin{frame}{Audio — Scheduled Playback \& Jitter Buffer}
\textbf{Local playback:}
\begin{itemize}
    \item Key press schedules \texttt{play\_note} and broadcast after \textbf{30 ms} (\texttt{PLAYOUT\_DELAY}).
    \item Release uses the same delay so local and remote notes stay aligned.
\end{itemize}

\vspace{0.3cm}
\textbf{JitterBuffer} for remote notes:
\begin{itemize}
    \item Priority queue keyed by a \textbf{monotonic deadline} (immune to wall-clock steps).
    \item Waits until the sender-scheduled play time, then plays or stops the note.
    \item Deduplicates by \texttt{(sender\_ip, seq)}; drops events more than 200 ms overdue.
    \item Before first sync, offset is 0 --- notes play $\sim$30 ms after arrival (graceful fallback).
\end{itemize}

\vspace{0.3cm}
\textbf{AudioEngine:} pygame mixer, 32 channels, 200 ms fadeout, 3 s stuck-note watchdog.
\end{frame}

\begin{frame}{User Interface}
\textbf{LobbyScreen:} player name, lobby name, host/join/refresh, discovered-lobby list.

\vspace{0.3cm}
\textbf{PianoScreen:} ASCII piano with box-drawing borders, player sidebar, activity log.

\vspace{0.3cm}
\textbf{Keyboard mapping:} \texttt{a s d f g h j k} (white keys), \texttt{w e t y u} (black keys).

\vspace{0.3cm}
\textbf{Input model:} Textual has no key-up events --- 500 ms timer auto-releases each note; key-repeat is debounced. Remote presses flash for 400 ms.
\end{frame}

\begin{frame}{Docker Deployment}
docker-compose provides a two-player LAN test environment:

\vspace{0.3cm}
\begin{itemize}
    \item Base image: \texttt{python:3.12-slim} with SDL2 runtime libraries.
    \item Custom bridge network \texttt{172.28.0.0/24} with fixed container IPs.
    \item Sounds pre-generated at image build time for instant startup.
    \item Requires \texttt{stdin\_open: true} and \texttt{tty: true} --- TUI apps cannot run detached.
\end{itemize}

\vspace{0.3cm}
\small
\texttt{docker compose build}\\
\texttt{docker compose run --rm player1} \textit{(terminal 1)}\\
\texttt{docker compose run --rm player2} \textit{(terminal 2)}
\end{frame}

\begin{frame}{Limitations \& Future Work}
\textbf{Current limitations:}
\begin{itemize}
    \item Hub topology: guests hear the host, but \textbf{not each other}.
    \item LAN-only discovery; no Internet/WAN support or NAT traversal.
    \item Fixed 500 ms auto-release approximates note-off rather than true key-up.
    \item Single octave (13 keys); no sustain pedal or velocity from key pressure.
\end{itemize}

\vspace{0.3cm}
\textbf{Possible extensions:}
\begin{itemize}
    \item Full-mesh or host-relay note forwarding for guest-to-guest audio.
    \item Wider keyboard range, MIDI input, and configurable instrument sounds.
\end{itemize}
\end{frame}
